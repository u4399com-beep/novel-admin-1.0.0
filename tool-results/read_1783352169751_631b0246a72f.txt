   172→// ─── Inline chapter editor panel ──────────────────────────────────────────────
   173→function ChapterEditorPanel({
   174→  chapter,
   175→  onClose,
   176→  onSaved,
   177→}: {
   178→  chapter: Chapter | null;
   179→  onClose: () => void;
   180→  onSaved: () => void;
   181→}) {
   182→  const [title, setTitle] = useState('');
   183→  const [content, setContent] = useState('');
   184→  const [saving, setSaving] = useState(false);
   185→  const [saveStatus, setSaveStatus] = useState<'idle' | 'saving' | 'saved'>('idle');
   186→  const autoSaveTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
   187→  const initialLoadRef = useRef(false);
   188→
   189→  // Load chapter content when selected
   190→  useEffect(() => {
   191→    if (!chapter) {
   192→      setTitle('');
   193→      setContent('');
   194→      setSaveStatus('idle');
   195→      initialLoadRef.current = false;
   196→      return;
   197→    }
   198→
   199→    // Fetch full chapter content
   200→    const loadChapter = async () => {
   201→      try {
   202→        const res = await fetch(`/api/chapters/${chapter.id}`);
   203→        if (res.ok) {
   204→          const data = await res.json();
   205→          setTitle(data.title);
   206→          setContent(data.content || '');
   207→          initialLoadRef.current = true;
   208→        }
   209→      } catch {
   210→        toast.error('加载章节内容失败');
   211→      }
   212→    };
   213→
   214→    loadChapter();
   215→  }, [chapter]);
   216→
   217→  // Auto-save debounce
   218→  const saveChapter = useCallback(
   219→    async (newTitle: string, newContent: string) => {
   220→      if (!chapter || !initialLoadRef.current) return;
   221→      if (saving) return;
   222→
   223→      setSaving(true);
   224→      setSaveStatus('saving');
   225→
   226→      try {
   227→        const wordCount = newContent.length;
   228→        const res = await fetch(`/api/chapters/${chapter.id}`, {
   229→          method: 'PUT',
   230→          headers: { 'Content-Type': 'application/json' },
   231→          body: JSON.stringify({
   232→            title: newTitle,
   233→            content: newContent,
   234→            wordCount,
   235→          }),
   236→        });
   237→
   238→        if (!res.ok) throw new Error('保存失败');
   239→
   240→        setSaveStatus('saved');
   241→        setTimeout(() => setSaveStatus('idle'), 2000);
   242→        onSaved();
   243→      } catch {
   244→        toast.error('自动保存失败');
   245→        setSaveStatus('idle');
   246→      } finally {
   247→        setSaving(false);
   248→      }
   249→    },
   250→    [chapter, saving, onSaved],
   251→  );
   252→
   253→  // Auto-save on content change
   254→  useEffect(() => {
   255→    if (!chapter || !initialLoadRef.current) return;
   256→
   257→    if (autoSaveTimerRef.current) {
   258→      clearTimeout(autoSaveTimerRef.current);
   259→    }
   260→
   261→    autoSaveTimerRef.current = setTimeout(() => {
   262→      saveChapter(title, content);
   263→    }, 1500);
   264→
   265→    return () => {
   266→      if (autoSaveTimerRef.current) {
   267→        clearTimeout(autoSaveTimerRef.current);
   268→      }
   269→    };
   270→  }, [title, content, chapter, saveChapter]);
   271→
   272→  const handleManualSave = async () => {
   273→    if (autoSaveTimerRef.current) {
   274→      clearTimeout(autoSaveTimerRef.current);
   275→    }
   276→    await saveChapter(title, content);
   277→  };
   278→
   279→  if (!chapter) return null;
   280→
   281→  const wordCount = content.length;
   282→  const charCount = content.replace(/\s/g, '').length;
   283→
   284→  return (
   285→    <div className="flex flex-col h-full">
   286→      {/* Editor header */}
   287→      <div className="flex items-center justify-between px-4 py-3 border-b bg-muted/30">
   288→        <div className="flex items-center gap-2 min-w-0">
   289→          <Type className="size-4 text-muted-foreground shrink-0" />
   290→          <Input
   291→            value={title}
   292→            onChange={(e) => setTitle(e.target.value)}
   293→            className="h-8 text-sm font-medium border-0 bg-transparent focus-visible:ring-0 px-1"
   294→            placeholder="章节标题"
   295→          />
   296→        </div>
   297→        <div className="flex items-center gap-2 shrink-0">
   298→          {/* Auto-save indicator */}
   299→          <div className="flex items-center gap-1.5 text-xs text-muted-foreground">
   300→            {saveStatus === 'saving' && (
   301→              <>
   302→                <Loader2 className="size-3 animate-spin" />
   303→                <span>保存中...</span>
   304→              </>
   305→            )}
   306→            {saveStatus === 'saved' && (
   307→              <>
   308→                <CheckCircle2 className="size-3 text-emerald-500" />
   309→                <span className="text-emerald-600 dark:text-emerald-400">已保存</span>
   310→              </>
   311→            )}
   312→            {saveStatus === 'idle' && (
   313→              <span className="tabular-nums">{wordCount.toLocaleString()} 字</span>
   314→            )}
   315→          </div>
   316→          <Separator orientation="vertical" className="h-4" />
   317→          <Button
   318→            variant="ghost"
   319→            size="icon"
   320→            className="size-8"
   321→            onClick={onClose}
   322→          >
   323→            <X className="size-4" />
   324→          </Button>
   325→        </div>
   326→      </div>
   327→
   328→      {/* Editor content */}
   329→      <div className="flex-1 overflow-hidden relative">
   330→        <Textarea
   331→          value={content}
   332→          onChange={(e) => setContent(e.target.value)}
   333→          className="absolute inset-0 resize-none rounded-none border-0 shadow-none focus-visible:ring-0 p-4 font-mono text-sm leading-loose min-h-full h-full"
   334→          placeholder="开始编写章节内容..."
   335→        />
   336→      </div>
   337→
   338→      {/* Editor footer */}
   339→      <div className="flex items-center justify-between px-4 py-2 border-t bg-muted/30 text-xs text-muted-foreground">
   340→        <div className="flex items-center gap-4">
   341→          <span>字数: {wordCount.toLocaleString()}</span>
   342→          <span>字符 (不含空格): {charCount.toLocaleString()}</span>
   343→        </div>
   344→        <Button
   345→          size="sm"
   346→          onClick={handleManualSave}
   347→          disabled={saving}
   348→          className="h-7 text-xs"
   349→        >
   350→          <Save className="size-3" />
   351→          手动保存
   352→        </Button>
   353→      </div>
   354→    </div>
   355→  );
   356→}

   359→export default function NovelDetailView() {

   381→  const [deletingChapter, setDeletingChapter] = useState<Chapter | null>(null);
   382→  const [deleting, setDeleting] = useState(false);
   383→  const [selectedChapter, setSelectedChapter] = useState<Chapter | null>(null);
   384→
   385→  const sensors = useSensors(
   386→    useSensor(PointerSensor, {
   387→      activationConstraint: { distance: 8 },
   388→    }),
   389→  );
   390→
   391→  // Fetch novel details
   392→  const fetchNovel = useCallback(async () => {
   393→    if (!selectedNovelId) return;
   394→    setLoadingNovel(true);
   395→    try {
   396→      const res = await fetch(`/api/novels/${selectedNovelId}`);
   397→      if (res.ok) {
   398→        const data = await res.json();
   399→        setNovel(data);
   400→      } else {
   401→        toast.error('小说不存在或已被删除');
   402→        setCurrentView('novels');
   403→      }
   404→    } catch {
   405→      toast.error('获取小说详情失败');
   406→    } finally {
   407→      setLoadingNovel(false);
   408→    }
   409→  }, [selectedNovelId, setCurrentView]);
   410→
   411→  // Fetch chapters
   412→  const fetchChapters = useCallback(async () => {
   413→    if (!selectedNovelId) return;
   414→    setLoadingChapters(true);
   415→    try {
   416→      const res = await fetch(`/api/novels/${selectedNovelId}/chapters`);
   417→      if (res.ok) {
   418→        const data = await res.json();
   419→        setChapters(data);
   420→      }
   421→    } catch {
   422→      toast.error('获取章节列表失败');
   423→    } finally {
   424→      setLoadingChapters(false);
   425→    }
   426→  }, [selectedNovelId]);
   427→
   428→  useEffect(() => {
   429→    fetchNovel();
   430→  }, [fetchNovel, refreshNovels]);
   431→
   432→  useEffect(() => {
   433→    fetchChapters();
   434→  }, [fetchChapters, refreshChapters]);
   435→
   436→  // ── Handlers ──────────────────────────────────────────────────────────────
   437→
   438→  const handleBack = () => {
   439→    setSelectedNovelId(null);
   440→    setSelectedNovel(null);
   441→    setCurrentView('novels');
   442→  };
   443→
   444→  const handleEditNovel = () => {
   445→    if (!novel) return;
   446→    setEditingNovel(novel);
   447→    setNovelFormOpen(true);
   448→  };
   449→
   450→  const handleDeleteNovel = async () => {
   451→    if (!novel) return;
   452→    setDeleting(true);
   453→    try {
   454→      const res = await fetch(`/api/novels/${novel.id}`, { method: 'DELETE' });
   455→      if (!res.ok) {
   456→        const data = await res.json();
   457→        throw new Error(data.error || '删除失败');
   458→      }
   459→      toast.success('小说已删除');
   460→      triggerRefreshNovels();
   461→      triggerRefreshDashboard();
   462→      setCurrentView('novels');
   463→      setSelectedNovelId(null);
   464→      setSelectedNovel(null);
   465→    } catch (error) {
   466→      toast.error(error instanceof Error ? error.message : '删除小说失败');
   467→    } finally {
   468→      setDeleting(false);
   469→      setDeleteNovelOpen(false);
   470→    }
   471→  };
   472→
   473→  const handleNewChapter = () => {
   474→    setEditingChapter(null);
   475→    setChapterFormOpen(true);
   476→  };
   477→
   478→  const handleEditChapter = (ch: Chapter) => {
   479→    setEditingChapter(ch);
   480→    setChapterFormOpen(true);
   481→  };
   482→
   483→  const handleDeleteChapterClick = (ch: Chapter) => {
   484→    setDeletingChapter(ch);
   485→    setDeleteChapterOpen(true);
   486→  };
   487→
   488→  const handleDeleteChapter = async () => {
   489→    if (!deletingChapter) return;
   490→    setDeleting(true);
   491→    try {
   492→      const res = await fetch(`/api/chapters/${deletingChapter.id}`, {
   493→        method: 'DELETE',
   494→      });
   495→      if (!res.ok) {
   496→        const data = await res.json();
   497→        throw new Error(data.error || '删除失败');
   498→      }
   499→      toast.success('章节已删除');
   500→      if (selectedChapter?.id === deletingChapter.id) {
   501→        setSelectedChapter(null);
   502→      }
   503→      triggerRefreshChapters();
   504→      triggerRefreshNovels();
   505→    } catch (error) {
   506→      toast.error(error instanceof Error ? error.message : '删除章节失败');
   507→    } finally {
   508→      setDeleting(false);
   509→      setDeleteChapterOpen(false);
   510→      setDeletingChapter(null);
   511→    }
   512→  };
   513→
   514→  const handleDragEnd = async (event: DragEndEvent) => {
   515→    const { active, over } = event;
   516→    if (!over || active.id === over.id) return;
   517→
   518→    const oldIndex = chapters.findIndex((c) => c.id === active.id);
   519→    const newIndex = chapters.findIndex((c) => c.id === over.id);
   520→    if (oldIndex === -1 || newIndex === -1) return;
   521→
   522→    const reordered = arrayMove(chapters, oldIndex, newIndex);
   523→    setChapters(reordered);
   524→
   525→    // Update sort orders
   526→    try {
   527→      await Promise.all(
   528→        reordered.map((ch, idx) =>
   529→          fetch(`/api/chapters/${ch.id}`, {
   530→            method: 'PUT',
   531→            headers: { 'Content-Type': 'application/json' },
   532→            body: JSON.stringify({ sortOrder: idx + 1 }),
   533→          }),
   534→        ),
   535→      );
   536→    } catch {
   537→      toast.error('排序更新失败');
   538→      fetchChapters();
   539→    }
   540→  };
   541→
   542→  const handleMoveChapter = async (chapterId: string, direction: 'up' | 'down') => {
   543→    const idx = chapters.findIndex((c) => c.id === chapterId);
   544→    if (idx === -1) return;
   545→    if (direction === 'up' && idx === 0) return;
   546→    if (direction === 'down' && idx === chapters.length - 1) return;
   547→
   548→    const swapIdx = direction === 'up' ? idx - 1 : idx + 1;
   549→    const reordered = arrayMove(chapters, idx, swapIdx);
   550→    setChapters(reordered);
   551→
   552→    try {
   553→      await Promise.all(
   554→        reordered.map((ch, i) =>
   555→          fetch(`/api/chapters/${ch.id}`, {
   556→            method: 'PUT',
   557→            headers: { 'Content-Type': 'application/json' },
   558→            body: JSON.stringify({ sortOrder: i + 1 }),
   559→          }),
   560→        ),
   561→      );
   562→    } catch {
   563→      toast.error('排序更新失败');
   564→      fetchChapters();
   565→    }
   566→  };
   567→
   568→  const handleChapterSaved = () => {
   569→    triggerRefreshChapters();
   570→    fetchNovel(); // refresh novel stats (word count)
   571→  };
   572→
   573→  // ── Loading state ─────────────────────────────────────────────────────────
   574→
   575→  if (loadingNovel) {
   576→    return (
   577→      <div className="flex-1 p-6 space-y-6">
   578→        <div className="flex items-center gap-4">
   579→          <Skeleton className="size-9" />
   580→          <Skeleton className="h-8 w-48" />
   581→        </div>
   582→        <div className="flex gap-6">
   583→          <Skeleton className="size-40 rounded-lg shrink-0" />
   584→          <div className="flex-1 space-y-3">
   585→            <Skeleton className="h-8 w-64" />
   586→            <Skeleton className="h-4 w-32" />
   587→            <Skeleton className="h-4 w-20" />
   588→            <Skeleton className="h-16 w-full" />
   589→          </div>
   590→        </div>
   591→      </div>
   592→    );
   593→  }
   594→
   595→  if (!novel) return null;
   596→
   597→  const statusInfo = statusMap[novel.status] || statusMap.ongoing;
   598→  const totalWords = chapters.reduce((sum, ch) => sum + ch.wordCount, 0);
   599→  const chapterCount = novel._count?.chapters ?? chapters.length;
   600→
   601→  return (
   602→    <div className="flex-1 flex flex-col overflow-hidden">
   603→      {/* ─── Header section ────────────────────────────────────────────── */}
   604→      <div className="p-6 pb-0">
   605→        {/* Back button */}
   606→        <Button
   607→          variant="ghost"
   608→          size="sm"
   609→          className="mb-4 -ml-2 text-muted-foreground"
   610→          onClick={handleBack}
   611→        >
   612→          <ArrowLeft className="size-4" />
   613→          返回小说列表
   614→        </Button>
   615→
